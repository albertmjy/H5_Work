function searchMostOccur(){
			var arr = ['a','x','b','d','m','a','k','m','p','j','a']
			
			var mostOccur = [];
			for (var i = 0; i < arr.length; i++) {
				var currentOccur = [arr[i]];
				for (var j = i; j < arr.length; j++) {
					if (arr[j] == currentOccur[0]){
						currentOccur.push(j);
					}
				}
				if(currentOccur.length > mostOccur.length){
					mostOccur = currentOccur
				}
				
			}
			
			var text = document.getElementById("disp");
			text.innerHTML = "The most occurred char is: " + mostOccur[0] + 
					", which appear " + (mostOccur.length - 1) + 
					". The positions are: " + mostOccur.slice(1) + "";
		}
